      >
        <div className={`absolute inset-0 bg-gradient-to-br ${currentBg}`} />

export default function App() {
  const [currentSlide, setCurrentSlide] = useState(0);
  const containerRef = useRef(null);
  const { scrollYProgress } = useScroll({ container: containerRef });
  const smoothProgress = useSpring(scrollYProgress, { stiffness: 100, damping: 30 });

  const handleScroll = () => {
    if (containerRef.current) {
      const scrollTop = containerRef.current.scrollTop;
      const height = window.innerHeight;
      const index = Math.round(scrollTop / height);
      if (index !== currentSlide && index >= 0 && index < SLIDES_DATA.length) {
        setCurrentSlide(index);
      }
    }
  };

  const scrollToSlide = (index) => {
    if (containerRef.current) {
      containerRef.current.scrollTo({
        top: index * window.innerHeight,
        behavior: 'smooth'
      });
    }
  };

  const handleNext = () => {
    if (currentSlide < SLIDES_DATA.length - 1) scrollToSlide(currentSlide + 1);
  };

  const handlePrev = () => {
    if (currentSlide > 0) scrollToSlide(currentSlide - 1);
  };

  // Keyboard navigation
  useEffect(() => {
    const handleKeyDown = (e) => {
      if (e.key === 'ArrowDown' || e.key === 'ArrowRight') handleNext();
      if (e.key === 'ArrowUp' || e.key === 'ArrowLeft') handlePrev();
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [currentSlide]);

  const renderSlide = (slideData, index) => {
    const isActive = index === currentSlide;
    
    switch(slideData.type) {
      case 'cover': return <CoverSlide key={index} data={slideData} isActive={isActive} />;
      case 'specialization': return <SpecializationSlide key={index} data={slideData} isActive={isActive} />;
      case 'market-reality': return <MarketRealitySlide key={index} data={slideData} isActive={isActive} />;
      case 'client-pains': return <ClientPainsSlide key={index} data={slideData} isActive={isActive} />;
      case 'approach': return <ApproachSlide key={index} data={slideData} isActive={isActive} />;
      case 'outsourcing': return <OutsourcingSlide key={index} data={slideData} isActive={isActive} />;
      case 'landing': return <LandingSlide key={index} data={slideData} isActive={isActive} />;
      case 'feature': return <FeatureSlide key={index} data={slideData} isActive={isActive} />;
      case 'channels': return <ChannelsSlide key={index} data={slideData} isActive={isActive} />;
      case 'logic': return <LogicSlide key={index} data={slideData} isActive={isActive} />;
      case 'case': return <CaseSlide key={index} data={slideData} isActive={isActive} />;
      case 'analytics': return <AnalyticsSlide key={index} data={slideData} isActive={isActive} />;
      case 'steps': return <StepsSlide key={index} data={slideData} isActive={isActive} />;
      case 'contact': return <ContactSlide key={index} data={slideData} isActive={isActive} />;
      default: return null;
    }
  };

  return (
    <div className="bg-black text-white font-sans antialiased selection:bg-cyan-400 selection:text-black overflow-hidden">
      <AnimatedBackground currentSlide={currentSlide} />
      <ProgressBar current={currentSlide} total={SLIDES_DATA.length} />
      
      <div 
        ref={containerRef}
        onScroll={handleScroll}
        className="h-screen w-full overflow-y-scroll snap-y snap-mandatory scroll-smooth no-scrollbar"
        style={{ scrollbarWidth: 'none', msOverflowStyle: 'none' }}
      >
        {SLIDES_DATA.map((slide, index) => renderSlide(slide, index))}
      </div>

      {/* Navigation Removed as requested */}
      
      {/* Mobile Hint */}
      <motion.div 
        className="fixed bottom-24 left-1/2 -translate-x-1/2 text-white/20 text-xs md:hidden pointer-events-none"
        animate={{ y: [0, 10, 0], opacity: [0.5, 1, 0.5] }}
        transition={{ duration: 1.5, repeat: Infinity }}
      >
        <span className="flex items-center gap-2">
          Листайте <ChevronDown size={14} />
        </span>
      </motion.div>

      {/* Slide Counter */}
      <motion.div 
        className="fixed top-4 right-4 z-50 text-white/30 text-sm font-mono"
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 1 }}
      >
        {String(currentSlide + 1).padStart(2, '0')} / {String(SLIDES_DATA.length).padStart(2, '0')}
      </motion.div>
    </div>
  );
}
